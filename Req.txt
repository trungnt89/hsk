async function speak(text) {
    const voice = document.getElementById("voice-select").value;
    if (voice === "browser") {
        window.speechSynthesis.cancel();
        const u = new SpeechSynthesisUtterance(text); u.lang = "zh-CN";
        window.speechSynthesis.speak(u); return;
    }
    const url = voice === "google" ? `https://translate.google.com/translate_tts?ie=UTF-8&client=gtx&tl=zh-CN&q=${encodeURIComponent(text)}` : `https://hsk-gilt.vercel.app/api/tts?text=${encodeURIComponent(text)}&voice=${voice}`;
    return new Promise(res => { audio.src = url; audio.onended = res; audio.play().catch(() => res()); });
}
